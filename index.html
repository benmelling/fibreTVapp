        function switchTab(tabId, el) {

            // UI Reset

            document.querySelectorAll('.nav-item').forEach(i => i.classList.remove('active'));

            if(el) el.classList.add('active');

           

            // Tab Content

            document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));

            document.getElementById('tab-' + tabId).classList.add('active');

           

            // Handle Live BG

            if(tabId === 'home') document.body.classList.add('mode-home');

            else document.body.classList.remove('mode-home');

            // Handle Dynamic Linear Menu

            if(tabId === 'linear') {

                document.getElementById('nav-lin-sports').classList.add('visible');

                document.getElementById('nav-lin-stream').classList.add('visible');

            } else {

                document.getElementById('nav-lin-sports').classList.remove('visible');

                document.getElementById('nav-lin-stream').classList.remove('visible');

            }

            // Handle Dynamic Plus Menu

            if(tabId === 'plus') {

                document.getElementById('nav-plus-movies').classList.add('visible');

                document.getElementById('nav-plus-tv').classList.add('visible');

                document.getElementById('nav-plus-tv').classList.remove('visible');

            }

        }
