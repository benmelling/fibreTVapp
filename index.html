    <script>

        // Init: Focus sidebar

        document.querySelector('.menu-item').focus();

        // Keyboard Navigation Logic

        document.addEventListener('keydown', (e) => {

            const active = document.activeElement;

            if (e.key === 'Enter') {

                if (active.classList.contains('menu-item')) {

                    // Switch Tabs

                    document.querySelectorAll('.menu-item').forEach(i => i.classList.remove('active-tab'));

                    active.classList.add('active-tab');

                   

                    document.querySelectorAll('.tab-content').forEach(t => t.style.display = 'none');

                    const targetId = active.getAttribute('data-target');

                    document.getElementById(targetId).style.display = 'block';

                   

                    // Focus first item in new tab

                    const firstItem = document.getElementById(targetId).querySelector('.focusable');

                    if(firstItem) firstItem.focus();

                }

            }

            // Arrow keys handle focus naturally via tabindex,

            // but we add styling listener below

        });

        // Add 'focused' class for styling when tab-navigating

        document.addEventListener('focusin', (e) => {

            document.querySelectorAll('.focused').forEach(el => el.classList.remove('focused'));

            if(e.target.classList.contains('focusable')) {

                e.target.classList.add('focused');

                e.target.scrollIntoView({behavior: "smooth", block: "center"});

            }

        });

    </script>
